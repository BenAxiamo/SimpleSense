242 struct __file {
243  char *buf; /* buffer pointer */
244  unsigned char unget; /* ungetc() buffer */
245  uint8_t flags; /* flags, see below */
246 #define __SRD 0x0001 /* OK to read */
247 #define __SWR 0x0002 /* OK to write */
248 #define __SSTR 0x0004 /* this is an sprintf/snprintf string */
249 #define __SPGM 0x0008 /* fmt string is in progmem */
250 #define __SERR 0x0010 /* found error */
251 #define __SEOF 0x0020 /* found EOF */
252 #define __SUNGET 0x040 /* ungetc() happened */
253 #define __SMALLOC 0x80 /* handle is malloc()ed */
254 #if 0
255 /* possible future extensions, will require uint16_t flags */
256 #define __SRW 0x0100 /* open for reading & writing */
257 #define __SLBF 0x0200 /* line buffered */
258 #define __SNBF 0x0400 /* unbuffered */
259 #define __SMBF 0x0800 /* buf is from malloc */
260 #endif
261  int size; /* size of buffer */
262  int len; /* characters read or written so far */
263  int (*put)(char, struct __file *); /* function to write one char to device */
264  int (*get)(struct __file *); /* function to read one char from device */
265  void *udata; /* User defined and accessible data. */
266 };
