299 void istream::getNumber(T* value) {
300  uint32_t tmp;
301  if ((T)-1 < 0) {
302  // number is signed, max positive value
303  uint32_t const m = ((uint32_t)-1) >> (33 - sizeof(T) * 8);
304  // max absolute value of negative number is m + 1.
305  if (getNumber(m, m + 1, &tmp)) {
306  *value = (T)tmp;
307  }
308  } else {
309  // max unsigned value for T
310  uint32_t const m = (T)-1;
311  if (getNumber(m, m, &tmp)) {
312  *value = (T)tmp;
313  }
314  }
