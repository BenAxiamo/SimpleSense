340 __ATTR_CONST__ static inline int isfinite (double __x)
341 {
342  unsigned char __exp;
343  __asm__ (
344  "mov %0, %C1 \n\t"
345  "lsl %0 \n\t"
346  "mov %0, %D1 \n\t"
347  "rol %0 "
348  : "=r" (__exp)
349  : "r" (__x) );
350  return __exp != 0xff;
351 }
